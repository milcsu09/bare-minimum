    xs : **I64 = malloc(4 * 8);

    defer {
      free(*(xs + 0));

      free(*(xs + 3));
      free(xs);
    };

    *(xs + 0) = malloc(4 * 8);
    *(xs + 1) = malloc(4 * 8);
    *(xs + 2) = malloc(4 * 8);
    *(xs + 3) = malloc(4 * 8);

  };
