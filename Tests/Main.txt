strlen(s) : (*U8) -> U64;


my_strlen(s) : (*U8) -> U64 =
  {
    i := 0;

    while *s > 0 then
      {
        s = s + 1;
        i = i + 1;
      };

    i
  };


__main(argc, argv) : (I32, **U8) -> Void =
  {

    printf("%ld\n", my_strlen("Test"));

  };
