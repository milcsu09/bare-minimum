
print(x) : (F64) -> Void;

puts(x) : (*U8) -> Void;

malloc(n) : (U64) -> *Void;
free(p) : (*Void) -> Void;

__main() : () -> Void =
  {

    p : *U8 = malloc(27);

    i := 0;

    while i < 26 then
      {
        *(p + i) = 65 + i;
        i = i + 1;
      };

    *(p + 26) = 0;

    puts(p);

    free(p);

  };
