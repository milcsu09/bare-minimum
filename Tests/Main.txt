
printf(fmt) : (*U8, ...) -> Void;

alias [x : F64, y : F64] as Vec2;

__main(argc, argv) : (I32, **U8) -> Void =
  {
    v : Vec2 = [1.5, 2.5];

    printf("Vec2 [%g, %g]\n", v.x, v.y);

    p := &v;

    printf("*Vec2 [%g, %g]\n", *p.x, *p.y);

    *p.x = 10;
    *p.y = 20;

    printf("Vec2 [%g, %g]\n", v.x, v.y);
    printf("*Vec2 [%g, %g]\n", *p.x, *p.y);
  };
